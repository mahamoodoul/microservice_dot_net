\newglossaryentry{microservice}{
  name        = microservice,
  plural      = microservices,
  description = {A small, independently deployable service that focuses on a single business capability}
}

\newglossaryentry{service-mesh}{
  name        = {service mesh},
  description = {A dedicated infrastructure layer for handling service‑to‑service communication, security and observability}
}

\newglossaryentry{vault}{
  name        = {Vault},
  description = {A HashiCorp tool for secrets management, encryption as‑a‑service, and identity‑based access}
}

\newglossaryentry{devops}{
  name        = DevOps,
  description = {A set of practices that combines software development (Dev) and IT operations (Ops) to shorten the development lifecycle and provide continuous delivery with high software quality. }
}

\newglossaryentry{microsegmentation}{
 name        = Microsegmentation, 
 description = {
    A security technique that divides a network into isolated segments at the workload or application level to reduce the attack surface and prevent lateral movement by threats.}
}

\newglossaryentry{stride}{
 name        = STRIDE,
 description = {
    A threat modeling framework developed by Microsoft that categorizes security threats into six types: Spoofing, Tampering, Repudiation, Information Disclosure, Denial of Service, and Elevation of Privilege.}
}

\newglossaryentry{pasta}{
 name        = PASTA,
 description = {
    An acronym for Process for Attack Simulation and Threat Analysis, a risk-centric threat modeling methodology that guides organizations through seven stages to identify and mitigate security threats.}
}
